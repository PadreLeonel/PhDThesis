\begin{frame}
  \frametitle{Benchmark}
  \framesubtitle{Setup}
  \monocolumn{
    \begin{enumerate}
    \item Random reference goodness models
    \item Feedback generated  from reference models using a Bernoulli process
    \item Goodness estimations generated for different $\#$ of feedback observations
    \end{enumerate}
    \vspace{1em}
    \begin{description}
    \item[Estimation Error] Absolute difference of the reference and estimated goodness values \\[1em]
    \item [Constant estimator] $\supergj = \frac{|\FB_{pass}|}{|\FB_{pass}| + |\FB_{fail}|}$
    \end{description}
  }
\end{frame}
